<meta charset="utf-8">
<script>
var money = parseInt(prompt('Сумма вклада:'));
var interest = parseInt(prompt('Процентная ставка, %:'));
var years = parseInt(prompt('Срок вклада, лет:'));
for (var i=1; i<=years; i++){
    money = money + money * interest / 100;
	alert('Прошло лет:' + i + ',сумма на счету:' + Math.floor(money));
}
</script>
